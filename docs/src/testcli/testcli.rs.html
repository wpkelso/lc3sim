1//! CLI for pure LC3 testing purposes.
2
3use std::{
4    env::args,
5    fs::File,
6    io::{stdin, BufReader},
7};
8
9use lc3sim_project::{
10    defs::{LC3Word, RegAddr},
11    executors::{core::CoreLC3, populate_from_bin, LC3},
12};
13
14fn main() {
15    let mut lc3 = CoreLC3::new();
16    for arg in args().skip(2) {
17        populate_from_bin(&mut lc3, BufReader::new(File::open(arg).unwrap()));
18    }
19
20    let start_line = LC3Word::from_str_radix(&args().nth(1).unwrap(), 16).unwrap();
21    while lc3.pc() != start_line {
22        lc3.step().unwrap()
23    }
24
25    loop {
26        for reg in 0..8 {
27            println!("Reg {reg}: {:#X}", lc3.reg(RegAddr::panic_from_u8(reg)));
28        }
29
30        println!(
31            "Current instruction {:#X}: {:?}",
32            lc3.pc(),
33            lc3.cur_inst().unwrap()
34        );
35
36        stdin().read_line(&mut String::new()).unwrap();
37        lc3.step().unwrap();
38    }
39}
