6/// Set the processor condition codes from `result`.
7pub fn set_condition_codes<P: LC3>(processor: &mut P, result: LC3Word) {
8    match (result as SignedLC3Word).cmp(&0) {
9        std::cmp::Ordering::Greater => processor.flag_positive(),
10        std::cmp::Ordering::Less => processor.flag_negative(),
11        std::cmp::Ordering::Equal => processor.flag_zero(),
12    }
13}

15/// Parses the opcode from a word.
16#[inline]
17pub const fn get_opcode(word: LC3Word) -> u8 {
18    // Opcode is always the top half byte
19    word.to_be_bytes()[0] >> 4

22/// Extracts a range of bits from a word.
23///
24/// `start`, `end` is inclusive.
25/// May panic or return undefined output if:
26/// * `start` >= LC3Word length
27/// * `end` >= LC3Word length
28/// * `end` > start
29///
30/// e.g. get_bits(0x00A2, 5, 1)
31/// * 0x00A2 -> 0000 0000 1010 0010
32/// * 0000 0000 10[10 001]0
33/// * return 0x0011 -> 1 0001
34#[inline]
35pub const fn get_bits(mut word: LC3Word, start: u8, end: u8) -> u16 {
36    const WORD_BITS: u8 = LC3Word::BITS as u8;
37    const WORD_SHIFT_START: u8 = WORD_BITS - 1;
38
39    debug_assert!(start < WORD_BITS);
40    debug_assert!(end <= start);
41
42    // Set bits above `start` to zero.
43    let shift_out_top = WORD_SHIFT_START - start;
44    word <<= shift_out_top;
45    word >>= shift_out_top;
46
47    // Remove bits below `end` to zero.
48    word >>= end;
49
50    // Only the given range remains in the word
51    word
52}

54/// Gets a single bit from a word.
55///
56/// May panic or return undefined output if
57/// `loc` >= LC3Word length
58///
59/// e.g. get_bit(0x00A2, 2)
60/// * 0x00A2 -> 0000 0000 1010 0010
61/// * 0000 0000 1010 0[0]10
62/// * return 0x0000
63#[inline]
64pub const fn get_bit(word: LC3Word, loc: u8) -> u8 {
65    get_bits(word, loc, loc) as u8
