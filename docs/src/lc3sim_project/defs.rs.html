58impl RegAddr {
59    /// `const` optimized u8 conversion.
60    ///
61    /// Panics when u8 is out of range.
62    pub const fn panic_from_u8(value: u8) -> Self {
63        match value {
64            0 => RegAddr::Zero,
65            1 => RegAddr::One,
66            2 => RegAddr::Two,
67            3 => RegAddr::Three,
68            4 => RegAddr::Four,
69            5 => RegAddr::Five,
70            6 => RegAddr::Six,
71            7 => RegAddr::Seven,
72            _ => panic!("Argument outside of [0, 7] (the valid LC-3 registers"),
73        }
74    }
75
76    /// Converts and calls [`Self::panic_from_u8`].
77    pub const fn panic_from_u16(value: u16) -> Self {
78        Self::panic_from_u8(value as u8)
79    }
80}

86impl TryFrom<u8> for RegAddr {
87    type Error = InvalidRegAddr;
88    fn try_from(value: u8) -> Result<Self, Self::Error> {
89        match value {
90            0 => Ok(RegAddr::Zero),
91            1 => Ok(RegAddr::One),
92            2 => Ok(RegAddr::Two),
93            3 => Ok(RegAddr::Three),
94            4 => Ok(RegAddr::Four),
95            5 => Ok(RegAddr::Five),
96            6 => Ok(RegAddr::Six),
97            7 => Ok(RegAddr::Seven),
98            x => Err(InvalidRegAddr(x)),
99        }
100    }
101}

103impl From<RegAddr> for u8 {
104    fn from(value: RegAddr) -> Self {
105        match value {
106            RegAddr::Zero => 0,
107            RegAddr::One => 1,
108            RegAddr::Two => 2,
109            RegAddr::Three => 3,
110            RegAddr::Four => 4,
111            RegAddr::Five => 5,
112            RegAddr::Six => 6,
113            RegAddr::Seven => 7,
114        }
115    }
116}
